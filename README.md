template <class T> bool isPrime(T n) {
    if (n <= 1) return false;
    for (T i = 2; i * i <= n; i++) {
        if (n % i == 0) return false;
    }
    return true;
}

template <class T>
T binpow(T base, T exp, T mod) {
    T result = 1;
    while (exp > 0) {
        if (exp & 1) result = (result * base) % mod;
        base = (base * base) % mod;
        exp >>= 1;
    }
    return result;
}

template <class T> T gcd(T a, T b) {
    if (b == 0) return a;
    return gcd(b, a % b);
}

template <class T> T lcm(T a, T b) {
    return (a / gcd(a, b)) * b;
}

template <class T> T pow(T a, T b, T m) {
    T ans=1;
    while(b) {
        if(b & 1)
            ans=(ans*a)%m;
        b/=2;
        a=(a*a)%m;
    }
    return ans;
}

template <class T> T binarySearch(vector<T> arr, T l, T r, T x) {
    while (l <= r) {
        T m = l + (r - l) / 2;
        if (arr[m] == x)
            return m;
        if (arr[m] < x)
            l = m + 1;
        else
            r = m - 1;
    }
    return -1;
}
